const SYSTEM_PROMPT = `You are a professional Quality Analyst specializing in sales call performance reviews.

Your task is to thoroughly analyze the attached call transcript and identify anything that could have been handled better, any missed opportunities, or any behaviors or patterns that may signal a performance concern.


For each issue or flag you identify, provide the following:

- Timestamp – Include the exact timestamp so the finding can be quickly verified

- Issue/Flag – Clearly describe what was said or done (or not said/done)

- Why It's a Concern – Explain why this deviates from best practice or the call guide

- Recommended Improvement – Suggest how the closer could have handled it better


Be thorough, objective, and specific. Do not generalize — every finding must be backed by a direct reference to the transcript. If a pattern repeats, flag it as a recurring concern and note all instances.


Use the Closer Sales Call Guide below as your primary benchmark:


NOTES: Inconsistencies in pricing is a leverage for urgency — offering discounts to push a deal faster is intentional.


The way we run sales calls:

- Conversational: These are corporate professionals. Have fun with it.

- Consultative: Educate them with brevity. Everyone is green unless they've bought a business before. Don't go too deep.

- We are not lawyers, CPAs, Lenders, or financial advisors.

- Challenger: If they say something wrong, challenge them.

- Avoid getting "friend zoned" — don't make it too casual or too far removed from a sales call.


FRAMEWORK:


Intro (2-5 min max): Light. How are you? Where are you calling from? Into the frame.


Frame: "Thanks for hopping on the call today. Just to give you an idea of how these calls go, first we will dive into more about you, what you are looking for from a business, and finally to see if we are a good fit and can actually help you based on your situation. If we can, I will dive into everything we do and answer any questions you have. Ready to get started?"


Discovery Questions:

1. Tell me about yourself and what brought you to the call? (Open-ended)

2. Out of curiosity why small business acquisition? (Gauge seriousness)

3. What is the plan with the job? (Level of commitment)

4. Did you have any partners? (DO NOT end the call to wait for a partner — do the call, send resources/recording, get both on a follow-up)

5. What types of industries are you looking at? (Criteria: Industry + Location + Cash Flow — specific in ONE only)

6. Are you looking to stay local or open to other locations? (Biggest lever if criteria too narrow)

8. What is the down payment you are working with? (Financially qualify)

9. Is that all cash or stocks/other sources?

10. Are you interested in using investors?

11. What do you need from us to succeed? / Dig on their biggest challenge.

12. How soon are you looking to buy the business?

13. How soon are you looking to get started with us?

14. No vacations coming up? (Must be ready to start immediately)

let files = [];


const dropzone = document.getElementById('dropzone');

const fileInput = document.getElementById('fileInput');

const auditBtn = document.getElementById('auditBtn');


dropzone.addEventListener('dragover', e => { e.preventDefault(); dropzone.classList.add('drag'); });

dropzone.addEventListener('dragleave', () => dropzone.classList.remove('drag'));

dropzone.addEventListener('drop', e => {

  e.preventDefault(); dropzone.classList.remove('drag');

  addFiles(Array.from(e.dataTransfer.files));

});

fileInput.addEventListener('change', e => {

  addFiles(Array.from(e.target.files));

  fileInput.value = '';

});


function addFiles(newFiles) {

  const allowed = ['.txt','.pdf','.vtt'];

  newFiles.forEach(f => {

    const ext = '.' + f.name.split('.').pop().toLowerCase();

    if (!allowed.includes(ext)) return;

    if (!files.find(x => x.name === f.name && x.size === f.size)) files.push(f);

  });

  renderFileList();

}


function removeFile(i) { files.splice(i,1); renderFileList(); }


function renderFileList() {

  const el = document.getElementById('fileList');

  if (!files.length) { el.innerHTML=''; auditBtn.disabled=true; return; }

  el.innerHTML = files.map((f,i) => `

    <div class="file-item">

      <span class="fname">📄 ${esc(f.name)}</span>

      <span class="fmeta">${(f.size/1024).toFixed(1)} KB</span>

      <button class="rm" onclick="removeFile(${i})">✕</button>

    </div>`).join('');

  auditBtn.disabled = false;

}


function readFile(f) {

  return new Promise((res,rej) => {

    const ext = f.name.split('.').pop().toLowerCase();

    if (ext === 'pdf') {

      const r = new FileReader();

      r.onload = () => res({ type:'pdf', data: r.result.split(',')[1] });

      r.onerror = () => rej(new Error('Failed to read PDF'));

      r.readAsDataURL(f);

    } else {

      const r = new FileReader();

      r.onload = () => res({ type:'text', data: r.result });

      r.onerror = () => rej(new Error('Failed to read file'));

      r.readAsText(f);

    }

  });

}


function parseVTT(raw) {

  const lines = raw.split('\n');

  const out = [];

  let ts = '';

  for (let l of lines) {

    l = l.trim();

    if (l.includes('-->')) { ts = l.split('-->')[0].trim(); continue; }

    if (l && !l.startsWith('WEBVTT') && !l.match(/^\d+$/)) {

      out.push(ts ? `[${ts}] ${l}` : l);

      ts = '';

    }

  }

  return out.join('\n');

}


async function auditFile(f, name) {

  const fileData = await readFile(f);

  const ext = f.name.split('.').pop().toLowerCase();

  let textContent = fileData.data;

  if (ext === 'vtt') textContent = parseVTT(fileData.data);


  let userContent;

  if (fileData.type === 'pdf') {

    userContent = [

      { type:'document', source:{ type:'base64', media_type:'application/pdf', data:fileData.data } },

      { type:'text', text:`Please audit this call transcript. The closer's name is ${name}. File: ${f.name}` }

    ];

  } else {

    userContent = `Please audit this call transcript. The closer's name is ${name}. File: ${f.name}\n\n---TRANSCRIPT---\n\n${textContent}`;

  }


  const res = await fetch('/api/audit', {

    method: 'POST',

    headers: { 'Content-Type': 'application/json' },

    body: JSON.stringify({

      model: 'claude-sonnet-4-20250514',

      max_tokens: 1000,

      system: SYSTEM_PROMPT,

      messages: [{ role:'user', content: userContent }]

    })

  });


  const data = await res.json();

  if (data.error) throw new Error(data.error.message || JSON.stringify(data.error));

  return data.content?.map(b => b.text||'').join('\n') || 'No response received.';

}


async function runAudit() {

  if (!files.length) return;

  auditBtn.disabled = true;

  auditBtn.textContent = `⏳ Auditing ${files.length} file${files.length>1?'s':''}...`;

  document.getElementById('errorBox').style.display = 'none';

  document.getElementById('reportsWrap').innerHTML = '';


  const name = document.getElementById('closerName').value.trim() || 'the closer';


  const pw = document.getElementById('progressWrap');

  pw.innerHTML = `<div class="progress-wrap">${files.map((f,i)=>`

    <div class="progress-item" id="pi-${i}">

      <span class="pi-name">📄 ${esc(f.name)}</span>

      <span class="badge pending" id="badge-${i}">Pending</span>

    </div>`).join('')}</div>`;


  const reports = [];

  for (let i = 0; i < files.length; i++) {

    setBadge(i,'running','⏳ Auditing...');

    try {

      const report = await auditFile(files[i], name);

      reports.push({ name: files[i].name, report });

      setBadge(i,'done','✅ Done');

    } catch(err) {

      reports.push({ name: files[i].name, report:null, error: err.message });

      setBadge(i,'fail','❌ Failed');

    }

  }


  const rw = document.getElementById('reportsWrap');

  rw.innerHTML = '';

  reports.forEach((r,i) => {

    const sec = document.createElement('div');

    sec.className = 'report-section';

    sec.innerHTML = `

      <div class="report-title">

        <h2>📋 ${esc(r.name)}</h2>

        ${r.report ? `<button class="copy-btn" onclick="copyText(${i})">Copy</button>` : ''}

      </div>

      <div class="report-body" id="rb-${i}">

        ${r.report ? renderReport(r.report) : `<p style="color:#dc2626">⚠️ ${esc(r.error||'Unknown error')}</p>`}

      </div>`;

    rw.appendChild(sec);

  });


  window._reports = reports;

  rw.scrollIntoView({ behavior:'smooth' });

  auditBtn.disabled = false;

  auditBtn.textContent = '🔍 Run Audit';

}


function setBadge(i,cls,txt) {

  const b = document.getElementById(`badge-${i}`);

  if (b) { b.className=`badge ${cls}`; b.textContent=txt; }

}


function renderReport(text) {

  const lines = text.split('\n');

  let html = '', inFinding = false;

  for (let line of lines) {

    if (line.startsWith('## ')) { html+=`<h2>${esc(line.slice(3))}</h2>`; }

    else if (line.startsWith('### ')) { if(inFinding){html+='</div>';inFinding=false;} html+=`<h3>${esc(line.slice(4))}</h3>`; }

    else if (line.startsWith('**Finding #')) { if(inFinding)html+='</div>'; html+=`<div class="finding"><p><strong>${esc(line.replace(/\*\*/g,''))}</strong></p>`; inFinding=true; }

    else if (line.startsWith('- **')) { const p=line.replace('- **','').split(':** '); html+=`<p><strong>${esc(p[0])}:</strong> ${esc(p.slice(1).join(': '))}</p>`; }

    else if (line.startsWith('**')&&line.endsWith('**')) { html+=`<p><strong>${esc(line.replace(/\*\*/g,''))}</strong></p>`; }

    else if (line==='---') { if(inFinding){html+='</div>';inFinding=false;} html+='<hr/>'; }

    else if (!line.trim()) { html+='<br/>'; }

    else { html+=`<p>${esc(line)}</p>`; }

  }

  if(inFinding) html+='</div>';

  return html;

}


function copyText(i) {

  const r = window._reports?.[i];

  if (!r?.report) return;

  navigator.clipboard.writeText(r.report).then(() => {

    const btns = document.querySelectorAll('.copy-btn');

    btns[i].textContent='Copied!';

    setTimeout(()=>btns[i].textContent='Copy',2000);

  });

}


function esc(s) {

  if(!s)return'';

  return s.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');

}
